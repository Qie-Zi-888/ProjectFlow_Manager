```sql
-- 验证数据关联
SELECT 
    p.project_id,
    p.project_name,
    p.team_id,
    pm.team_name
FROM project p
LEFT JOIN project_member pm ON p.team_id = pm.team_id AND pm.project_id = p.project_id
WHERE p.team_id IS NOT NULL;
```
